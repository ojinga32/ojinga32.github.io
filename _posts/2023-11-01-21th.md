# 자바스크립트 오브젝트

```
<script>

        /*
            Objects
            one of the JavaScript's data types.
            a collection of related data and/or functionality.
            nearly all objects in JavaScript are instances of Object
            Object = {key : value};
        */

        // 1. Literals and properties
        const obj1 = {};                    // 'object literal' syntax
        const obj2 = new Object();          // 'object constructor' syntax 

        function print(person) {
            console.log(person.name);
            console.log(person.age);
        }

        const eojin = { name: 'eojin', age: 22};
        print(eojin);

        eojin.job = 'programmer';
        console.log(eojin.job);

        delete eojin.job;

        // 2. Computed properties
        // key should be alway String
        console.log(eojin.name);
        console.log(eojin['age']);      // [age] (X) || ['age'] (O)
        eojin['job'] = 'programmer'
        console.log(eojin.job);

        function printValue(obj, key) {
            console.log(obj[key]);
        }
        printValue(eojin, 'name');
        printValue(eojin, 'age');
        
        // 3. Property value shorthand
        const person1 = {name: 'star', age: 999};
        const person2 = {name: 'sponge', age: 11};
        const person3 = {name: 'crab', age: 23};
        const person4 = new Person('ramen', 90);
        console.log(person4);

        // 4. COnstructor function
        function Person(name, age) {
            // this = {};
                this.name = name;
                this.age = age;
            // return this;
        }

        // 5. in Operator: property existence check (key in obj);
        console.log('name' in eojin);
        console.log('job' in eojin);
        console.log('random' in eojin);
        console.log(eojin.random);

        // 6. for..in vs for..of
        // for (key in obj) 
        for (key in eojin) {
            console.log(key);
        }

        // for (value of iterable)
        const array = [1, 2, 3, 4, 5];
        /*
            for(let i = 0; i<array.length; i++) {
            console.log(`i = ${i}`);
        }
        */
        for (value of array) {
            console.log(`${value}`);
        }

        // 7. Fun cloning
        // Object.assign(destm [obj1, obj2, obj3...])
        const user = {name: 'eojin', age: 22};
        const user2 = user;
        user2.name = 'changed';
        console.log(user);

        // old way
        const user3 ={};
        for (key in user) {
            user3[key] = user[key];
        }
        console.log(user3);

        console.clear();
        // new way
        const user4 = {};
        Object.assign(user4, user);
        const user5 = Object.assign({}, user);
        console.log(user4);
        console.log(user5);

        // another example
        const fruit1 = {color: 'red'};
        const fruit2 = {color: 'blue', size: 'huge'};
        const mixed = Object.assign({}, fruit1, fruit2);
        console.log(mixed);             // 동일한 값(color)이 있다면 뒤에 쓴 값을 덮어씌운다. 새로운 것은 추가
    </script>
```


1. for in은 객체 / for of는 배열
2. key는 무조건 string
3. ```
     function printValue(obj, key) {
            console.log(obj[key]);
            console.log(obj.key);
        }
          printValue(eojin, 'name');
   ```
   두가지의 차이점은 obj[key]는 obj.name 반면에 obj.key는 obj에 key라는 속성을 가진 값을 찾는다.
