# 연산자, 비교문, 반목문

```
<!DOCTYPE html>
<html lang="en">
<head>
    <meta charset="UTF-8">
    <meta name="viewport" content="width=device-width, initial-scale=1.0">
    <title>Document</title>
</head>
<body>
    <script>

        // 1. String concatenation
        console.log('my' + ' cat');     // 
        console.log('1' + 2);           // string
        console.log(`string literals: 3 + 6 = ${3 + 6}`);       // 변수 값을 계산해서 string으로 리턴
 
        // 2. Numeric operators
        console.log(1 + 1);     // add
        console.log(1 - 1);     // substract     
        console.log(1 / 1);     // divide
        console.log(1 * 1);     // multiply
        console.log(5 % 2);     // remainder  :  나머지 값
        console.log(2 ** 3);    // exponentiation : 제곱

        // 3. Increment and decrement operators
        let counter = 2;
        const preIncrement = ++counter;             // preIncrement = 2 + 1
        console.log(`preIncrement : ${preIncrement}, counter: ${counter}`);
        
        const postIncrement = counter++;             // postIncrement = 3 + 1
        console.log(`postIncrement: ${postIncrement}, counter: ${counter}`);

        // 4. Assignment operators
        let x = 3;
        let y = 6;
        x += y;     // x = x + y;
        x -= y;
        x *= y;
        x /= y;

        // 5. Comparison operators
        console.log(10 < 6);    // letss than
        console.log(10 <= 6);    // letss than or equal
        console.log(10 > 6);    // greater than
        console.log(10 >= 6);    // greater than equal

        // 6. Logical operators: || (or), && (and), ! (not)

        const value1 = false;
        const value2 = 4 < 2;

        // 3가지 조건충 하나만 충족
        console.log(`or: ${value1 || value2 || check()}`);     // 초보자를 위한 팁 : 확률이 가장 높은 것을 뒤에 놓아야 한다.
      // 초보자를 위한 팁 : 확률이 가장 높은 것을 뒤에 놓아야 한다.

        // 3가지 모두 충족
        console.log(`or: ${value1 && value2 && check()}`);      // 초보자를 위한 팁 : 확률이 가장 높은 것을 뒤에 놓아야 한다.

        // !false = true 
        console.log(!value1);

        function check() {
        for(let i=0; i<10; i++) {
            console.log("hello");
        }
            return true;
        }

        
        // 7. Equality
        const stringFive = '5';
        const numberFive = 5;

        // == loose equality, with type conversion
        console.log(stringFive == numberFive);
        console.log(stringFive != numberFive);

        // === strict equality, no type conversion
        console.log(stringFive === numberFive);
        console.log(stringFive !== numberFive);

        // object equality by reference
        const eojin1 = {name: 'eojin'};
        const eojin2 = {name: 'eojin'};
        const eojin3 = eojin1;

        console.log(eojin1 == eojin2);       // 참조 값이 달라서 false;
        console.log(eojin1 === eojin2);     // 참조 값이 달라서 false;
        console.log(eojin1 === eojin3);     // 참조 값이 같아서 true;

        console.log(0 == false);            // 0 = false                       true
        console.log(0 === false);           // but 0의 type은 boolean이 아니라  false
        console.log('' == false);           // 빈 문자열은 false                true
        console.log("" === false);          // null = false                    true
        console.log(null == false);         // null은 boolean이 아니라          true
        console.log(null === false);
        

        // 8. Conditional operators: if
        const name = 'eojin';
        if (name === 'eojin') {
            console.log("Welcome to Eojin");
        } else if (name === 'programmer') {
            console.log("프로그래머")
        } else {
            console.log("안녕 ㅎㅎ")
        }

        // 9. Ternary operator: ?                       이것은 간단한 비교문에만 사용하도록 
        // condition ? vlaue1 : valye2
        console.log(name === 'eojin' ? 'yes' : 'no');

        // 10. Switch statement
        // use for multiple if checks
        // use for enum-like value check
        // use for multiple type checks in TS

        const browser = 'novel';
        switch(browser) {
            case 'book':
                console.log('book');
                break;
            case 'note':
                console.log('note');
                break;
            case 'novel':
                console.log('novel');
                break;
            case 'pen':
            console.log('pen');
            break;
        }

        // 11. Loops
        // While loop, while the condition is truethy,
        // body code is executed,
        let i = 3;
        while (i>0) {
            console.log("hello");
            i--;
        }


        // do while loop, body code is executed, first,
        // then check the condition
        do {
            console.log(`i = ${i}`);
            i--;
        } while (i>3);
    </script>
</body>
</html>
```
